    var map = new gMap.Map('map',{
        layers:[
            'OSM'
            ],
            center:[100.83469116179398, 24.73574699993908],
            zoom:9
        });
        //图形图层，opacity表示图层透明度
        var graphicLayer = new gMap.layers.GraphicLayer({opacity:0.8});
        //地图加图层
        map.addLayer(graphicLayer);
        function drawCircle(){
                //画区域,传中心点，半径，半径单位为米
                var circle = new gMap.geometry.Circle([100.83469116179398, 24.73574699993908],10000);
                //线样式，gMap.symbols.SimpleLineSymbol.STYLE_SOLID为常量，表示线的样式为solid，第二个参数为颜色，第三个参数为线的宽度，单位px
                /*
                gMap.symbols.SimpleLineSymbol.STYLE_SOLID可替换为：自己试试吧，不注释了
                gMap.symbols.SimpleLineSymbol.STYLE_DASH 虚线
                gMap.symbols.SimpleLineSymbol.STYLE_DASHDOT 虚线点结合
                gMap.symbols.SimpleLineSymbol.STYLE_DASHDOTDOT
                gMap.symbols.SimpleLineSymbol.STYLE_DOT
                gMap.symbols.SimpleLineSymbol.STYLE_LONGDASH
                gMap.symbols.SimpleLineSymbol.STYLE_LONGDASHDOT
                gMap.symbols.SimpleLineSymbol.STYLE_NULL
                gMap.symbols.SimpleLineSymbol.STYLE_SHORTDASH
                gMap.symbols.SimpleLineSymbol.STYLE_SHORTDASHDOT
                gMap.symbols.SimpleLineSymbol.STYLE_SHORTDASHDOTDOT
                gMap.symbols.SimpleLineSymbol.STYLE_SHORTDOT
                gMap.symbols.SimpleLineSymbol.STYLE_SOLID
                */
                var lineSymbol = new gMap.symbols.SimpleLineSymbol(gMap.symbols.SimpleLineSymbol.STYLE_SOLID,'red',4);
                //填充样式，gMap.symbols.SimpleFillSymbol.STYLE_SOLID为常量表示为颜色填充，第二个参数为线样式，，第三个参数为填充色
                /*gMap.symbols.SimpleFillSymbol.STYLE_SOLID可替换为:
                gMap.symbols.SimpleFillSymbol.STYLE_BACKWARD_DIAGONAL(正斜线填充)
                gMap.symbols.SimpleFillSymbol.STYLE_CROSS(方格填充)
                gMap.symbols.SimpleFillSymbol.STYLE_DIAGONAL_CROSS(斜方格填充)
                gMap.symbols.SimpleFillSymbol.STYLE_FORWARD_DIAGONAL(反斜线填充)
                gMap.symbols.SimpleFillSymbol.STYLE_HORIZONTAL(横线填充)
                gMap.symbols.SimpleFillSymbol.STYLE_VERTICAL(竖线填充)
                */
                var fillSymbol = new gMap.symbols.SimpleFillSymbol(gMap.symbols.SimpleFillSymbol.STYLE_SOLID,lineSymbol,'#FFCCCC');
                //图形，第一个参数为geometry，第二个参数为symbols
                var graphic =  new gMap.Graphic(circle,fillSymbol);
            
                //图层加图形
                graphicLayer.add(graphic);
            
        }

        function clearCircle(){
                graphicLayer.clear()
        }
                    
                                <div>
                                <button onclick="drawCircle()">开始画圆</button>
                                <button onclick="clearCircle()">清除元素</button>
